--***Original Postgress service env format:***--
PSQL_HOST="DEPLOYED PUBLIC URL NOT INCLUDING HTTP"
PSQL_USER="USERNAME"
PSQL_PASSWORD="PASSWORD"
PSQL_DATABASENAME="DBNAME"
PORT="PORT NUMBER"

--***MariaDB 10mil env format:***--
MDB_URL="URL OF HOSTED DATABASE / DEPLOYED OR LOCAL"
MDB_USERNAME="USERNAME TO CONNECT TO DB"
MDB_PASS="PASSWORD OF USER"
MDB_DBNAME="NAME OF DB BEING USED"
PORT="PORT"
